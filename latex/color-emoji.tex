% !Mode:: "TeX:UTF-8:Main"
% Date: 13.05.2018
% Description: demonstrates how use and to color emoji fonts
% Ressources: the used emoji fonts
% Links: https://tex.stackexchange.com/a/387069/2388
% for low-level command see the plain example
\documentclass{article}

\usepackage{fontspec}
\directlua{fonts.handlers.otf.registerpalette("demo", {
        { g = 0 },
        { g =0.5 },
        { r = 1  },
        { r = 1, b= 1}, %value used by the emoji
    }
    )}

\newfontface\Emojinature{seguiemj.ttf}[RawFeature={colr=yes}]
\newfontface\Emojicolor {seguiemj.ttf}[RawFeature={colr=demo}]
\begin{document}
%Default colors:
     \Emojinature \char"2661 \char"1F34D \char"1F986

% own colors:
     \Emojicolor \char"2661 \char"1F34D \char"1F986

\end{document}
