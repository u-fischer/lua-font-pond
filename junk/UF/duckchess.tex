% Springer %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcommand{\springer}[1]{%
	\begin{tikzpicture}
		\begin{pgfinterruptboundingbox}
			\begin{scope}[yshift=-6]
				\clip[rotate=-5] (0.68,2.38) ellipse (0.3 and 0.4);
				\fill[#1,rotate=-5](0.28,2.26)ellipse (0.3 and 0.4);
			\end{scope}
		\end{pgfinterruptboundingbox}
		\duck[body=#1,bill=accent,horsetail,mohican=accent]
\path (1.06,2.2)--++(0,0.15);
		\begin{pgfinterruptboundingbox}
			\begin{scope}[yshift=-5,xshift=1]
				\clip[rotate=-5] (0.68,2.38) ellipse (0.3 and 0.4);
				\fill[#1,rotate=-5](1.06,2.2) ellipse (0.3 and 0.4);
			\end{scope}
		\end{pgfinterruptboundingbox}
		\ifcutlines
			\draw[cutline] (1.,0.1) -- (1.,0.5);
		\fi		
	\end{tikzpicture}
}

% Dame %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcommand{\dame}[1]{%

		\duck[body=#1,necklace=gray,longhair=accent,bill=accent]
